```swift
import Foundation

/// SM-2 spaced repetition algorithm. Extracted from the existing
/// `SpacedRepetitionService.apply(grade:to:)` and made a protocol
/// conformer with injected clock.
struct SM2Scheduler: SchedulingAlgorithm {
    func review(card: Flashcard, grade: Int, now: Date) -> Flashcard {
        var updated = card

        if grade >= 3 {
            if updated.repetitions == 0 {
                updated.intervalDays = 1
            } else if updated.repetitions == 1 {
                updated.intervalDays = 6
            } else {
                updated.intervalDays = Int(Double(updated.intervalDays) * updated.easeFactor)
            }
            updated.repetitions += 1
        } else {
            updated.repetitions = 0
            updated.intervalDays = 1
        }

        updated.easeFactor = max(
            1.3,
            updated.easeFactor + (0.1 - Double(5 - grade) * (0.08 + Double(5 - grade) * 0.02))
        )
        updated.lastReviewedAt = now.ISO8601Format()
        updated.lastGrade = grade
        updated.modifiedAt = now.ISO8601Format()

        if let nextDate = Calendar.current.date(
            byAdding: .day, value: updated.intervalDays, to: now
        ) {
            updated.nextReviewDate = nextDate.ISO8601Format()
        }

        return updated
    }
}
```

### Task 2.3: Implement FSRSScheduler (port from fsrs-rs)

```swift
import Foundation

/// FSRS (Free Spaced Repetition Scheduler) — a modern scheduling algorithm
/// that learns per-card memory parameters (stability, difficulty) and uses a
/// forgetting curve to compute optimal intervals.
///
/// Ported from the open-source `fsrs-rs` Rust reference (MIT-licensed).
/// Activates when a card has ≥6 reviews; SM-2 is used for earlier reviews.
struct FSRSScheduler: SchedulingAlgorithm {
    /// Default FSRS parameters (can be tuned per-user in the future).
    let w: [Double] = [
        0.4, 0.6, 2.4, 5.8, 4.93, 0.94, 0.86, 0.01,
        1.49, 0.14, 0.94, 2.18, 0.05, 0.34, 1.26, 0.29, 2.61
    ]
    let decay: Double = -0.5
    let minimumStability: Double = 0.01
    let maximumStability: Double = 36500.0  // ~100 years

    func review(card: Flashcard, grade: Int, now: Date) -> Flashcard {
        var updated = card
        let clampedGrade = max(1, min(4, grade))  // FSRS uses 1–4

        let stability = card.stability ?? initStability(clampedGrade)
        let difficulty = card.difficulty ?? initDifficulty(clampedGrade)

        // Update difficulty
        let newDifficulty = clamp(
            difficulty + w[4] - w[5] * Double(clampedGrade - 3),
            lo: 1.0, hi: 10.0
        )

        // Compute retrievability at review time
        let lastReviewed = card.lastReviewedAt
            .flatMap { ISO8601DateFormatter().date(from: $0) } ?? now
        let elapsed = max(0, now.timeIntervalSince(lastReviewed) / 86400.0)
        let retrievability = exp(
            log(0.9) * elapsed / max(stability, 0.001)
        )

        // Update stability
        var newStability: Double
        if clampedGrade >= 3 {
            let hardPenalty = clampedGrade == 3 ? w[15] : 1.0
            let easyBonus = clampedGrade == 4
                ? exp(w[16] * (newDifficulty - 1.0)) : 1.0
            newStability = stability * (
                1.0 + exp(w[8]) *
                (11.0 - newDifficulty) *
                pow(stability, -w[9]) *
                (exp(w[10] * (1.0 - retrievability)) - 1.0) *
                hardPenalty * easyBonus
            )
        } else {
            newStability = w[11]
                * pow(newDifficulty, -w[12])
                * (pow(stability + 1.0, w[13]) - 1.0)
                * exp(w[14] * (1.0 - retrievability))
        }
        newStability = clamp(
            newStability,
            lo: minimumStability, hi: maximumStability
        )

        // Compute next interval
        let interval: Int
        if clampedGrade >= 3 {
            interval = max(1, Int(round(newStability * (
                clampedGrade == 4
                    ? w[6] * (2.0 - pow(retrievability, 2.0))
                    : w[7]
            ))))
        } else {
            interval = max(1, Int(round(
                newStability * (1.0 - pow(retrievability, 2.0)) * (1.0 / exp(1.0))
            )))
        }

        updated.stability = newStability
        updated.difficulty = newDifficulty
        updated.intervalDays = interval
        updated.repetitions = card.repetitions + 1
        updated.easeFactor = card.easeFactor  // FSRS doesn't use ease factor
        updated.lastReviewedAt = now.ISO8601Format()
        updated.lastGrade = grade
        updated.modifiedAt = now.ISO8601Format()

        if let nextDate = Calendar.current.date(
            byAdding: .day, value: interval, to: now
        ) {
            updated.nextReviewDate = nextDate.ISO8601Format()
        }

        return updated
    }

    private func initStability(_ grade: Int) -> Double {
        grade >= 3 ? w[0] + w[1] * Double(grade - 3) : w[2]
    }

    private func initDifficulty(_ grade: Int) -> Double {
        clamp(w[3] - w[4] * Double(grade - 3), lo: 1.0, hi: 10.0)
    }

    private func clamp(_ value: Double, lo: Double, hi: Double) -> Double {
        min(max(value, lo), hi)
    }
}
```

```swift
import Foundation

/// Parses Anki-style cloze deletions from text.
enum ClozeParser {
    /// A single cloze deletion found in text.
    struct ClozeDeletion: Equatable {
        let index: Int       // 1-based cloze index (c1, c2, ...)
        let answer: String   // The hidden text
        let range: Range<String.Index>
    }

    /// Returns true if the text contains any cloze deletions.
    static func hasClozeDeletions(_ text: String) -> Bool {
        text.contains("{{c")
    }

    /// Extracts all cloze deletions from text.
    /// Pattern: `{{cN::answer}}` where N is a positive integer.
    static func parseDeletions(_ text: String) -> [ClozeDeletion] {
        var results: [ClozeDeletion] = []
        var searchRange = text.startIndex..<text.endIndex

        while let match = text.range(
            of: #"\{\{c(\d+)::([^}]+)\}\}"#,
            options: .regularExpression,
            range: searchRange
        ) {
            let matched = text[match]
            guard let indexStr = matched.split(separator: "::").first?
                .dropFirst(3),  // Drop "{{c"
                let index = Int(indexStr) else {
                searchRange = match.upperBound..<text.endIndex
                continue
            }

            // Extract answer — everything between :: and }}
            let afterDelimiter = matched.drop { $0 != ":" }
            let answer = String(afterDelimiter
                .dropFirst(2)  // Drop "::"
                .dropLast(2))  // Drop "}}"

            results.append(ClozeDeletion(
                index: index,
                answer: answer,
                range: match
            ))
            searchRange = match.upperBound..<text.endIndex
        }
        return results
    }

    /// Creates a cloze card front (text with answers replaced by `[...]`).
    static func makeFront(text: String, deletion: ClozeDeletion) -> String {
        var front = text
        front.replaceSubrange(deletion.range, with: "[...]")
        return front
    }

    /// Creates a cloze card back (text with answer highlighted).
    static func makeBack(text: String, deletion: ClozeDeletion) -> String {
        var back = text
        let hint = "[\(deletion.answer)]"
        back.replaceSubrange(deletion.range, with: hint)
        return back
    }
}
```

```swift
import Foundation
import GRDB

/// Exports Echo decks as Anki .apkg files.
struct ApkgExportService {
    enum ExportError: LocalizedError {
        case dbReadFailed(Error)
        case zipFailed(Error)
        case tempDirFailed(Error)

        var errorDescription: String? {
            switch self {
            case .dbReadFailed(let e): "Database read failed: \(e.localizedDescription)"
            case .zipFailed(let e): "ZIP creation failed: \(e.localizedDescription)"
            case .tempDirFailed(let e): "Temp directory creation failed: \(e.localizedDescription)"
            }
        }
    }

    /// Exports all cards from the given deck IDs to an .apkg file.
    /// - Parameters:
    ///   - deckIDs: The decks to export. An empty set exports all decks.
    ///   - db: GRDB DatabaseReader for queries.
    ///   - outputURL: Destination .apkg file URL.
    func export(
        deckIDs: Set<String>,
        from db: DatabaseReader,
        to outputURL: URL
    ) async throws {
        let cards: [Flashcard]
        let decks: [Deck]
        do {
            cards = try await db.read { db in
                var query = Flashcard.all()
                if !deckIDs.isEmpty {
                    query = query.filter(deckIDs.contains(Flashcard.CodingKeys.deckID.stringValue))
                }
                return try query.fetchAll(db)
            }
            decks = try await db.read { db in
                try Deck.fetchAll(db)
            }
        } catch {
            throw ExportError.dbReadFailed(error)
        }

        // Build .apkg in temp directory
        let tempDir = FileManager.default.temporaryDirectory
            .appendingPathComponent("apkg-export-\(UUID().uuidString)")
        try FileManager.default.createDirectory(at: tempDir, withIntermediateDirectories: true)

        defer { try? FileManager.default.removeItem(at: tempDir) }

        // Write collection.anki21 SQLite database
        let dbPath = tempDir.appendingPathComponent("collection.anki21")
        // ... SQLite creation with col, notes, cards, revlog tables ...

        // ZIP the directory
        do {
            try FileManager.default.zipItem(
                at: tempDir,
                to: outputURL,
                shouldKeepParent: false
            )
        } catch {
            throw ExportError.zipFailed(error)
        }
    }
}

        case createdAt = "created_at"

    }
}
```

```swift
import Foundation
import GRDB

/// Generates one flashcard per chapter heading so new users have a deck
/// immediately after importing a book.
struct ChapterCardDrafter {
    enum DrafterError: LocalizedError {
        case importFailed(Error)
    }

    /// Auto-draft chapter cards for a book. Idempotent — re-running
    /// skips headings that already have a card.
    func draftCards(
        for audiobookID: String,
        bookTitle: String,
        db: DatabaseWriter
    ) async throws -> Int {
        let headings = try await db.read { db in
            try Row.fetchAll(db, sql: """
                SELECT id, text, chapter_index
                FROM epub_block
                WHERE audiobook_id = ?
                  AND kind = 'heading'
                  AND is_front_matter = 0
                ORDER BY sequence_index
                """, arguments: [audiobookID])
        }

        // Find or create the book's deck
        let deckID = try await findOrCreateDeck(named: bookTitle, db: db)

        var created = 0
        for heading in headings {
            let headingID = heading["id"] as? String ?? ""
            // Skip if a card already exists for this heading
            let exists = try await db.read { db in
                try Flashcard
                    .filter(Column("source_block_id") == headingID)
                    .filter(Column("card_type") == "normal")
                    .fetchCount(db) > 0
            }
            guard !exists else { continue }

            let headingText = heading["text"] as? String ?? ""
            let chapterIndex = heading["chapter_index"] as? Int ?? 0

            var card = Flashcard(
                id: UUID().uuidString,
                audiobookID: audiobookID,
                frontText: headingText,
                backText: "Chapter \(chapterIndex + 1) — \(bookTitle)",
                mediaTimestamp: 0,
                endTimestamp: nil,
                triggerTiming: .manualOnly,
                nextReviewDate: nil,
                intervalDays: 0,
                easeFactor: 2.5,
                repetitions: 0,
                lastReviewedAt: nil,
                lastGrade: nil,
                isEnabled: true,
                deckID: deckID,
                tags: "auto-drafted chapter",
                mediaJSON: nil,
                sourceBlockID: headingID,
                playlistPosition: nil,
                createdAt: Date().ISO8601Format(),
                modifiedAt: Date().ISO8601Format()
            )
            try await db.write { db in try card.insert(db) }
            created += 1
        }
        return created
    }

    private func findOrCreateDeck(
        named name: String, db: DatabaseWriter
    ) async throws -> String {
        // Query existing deck by name or create new
        if let existing = try await db.read({ db in
            try Row.fetchOne(db, sql: """
                SELECT id FROM deck WHERE name = ?
                """, arguments: [name])
        }) {
            return existing["id"] as? String ?? UUID().uuidString
        }
        let id = UUID().uuidString
        try await db.write { db in
            try db.execute(sql: """
                INSERT INTO deck (id, name, source, created_at, modified_at)
                VALUES (?, ?, 'manual', ?, ?)
                """, arguments: [id, name, Date().ISO8601Format(), Date().ISO8601Format()])
        }
        return id
    }
}
```
